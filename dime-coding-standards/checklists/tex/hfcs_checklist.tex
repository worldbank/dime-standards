\section*{High-Frequency Checks (HFCs)}
This checklist focuses on reviewing the \textbf{key outputs of high-frequency checks} to ensure data quality. It highlights critical components of HFCs, but some sections may not apply to the specific checks you are reviewing. If the code also includes cleaning, construction, or analysis, please refer to the respective checklists \href{https://github.com/worldbank/dime-standards/tree/master/dime-coding-standards/checklists}{here}.

\noindent Once completed, please submit this checklist as an attachment along with \href{https://survey.wb.surveycto.com/collect/code_review_summary?caseid=}{this form}.

\section*{General}
\begin{itemize}
    \setlength\itemsep{-0.1em}
    \item[] \dimeCheckBox{The data shared does not contain any personally identifying information (PII).}
   \item[] \dimeCheckBox {All scripts run from the main script after adding the correct folder path to line(s) X (and XX)}
    \item[] \dimeCheckBox{The code generates a tracking sheet or dashboard to monitor survey progress.}
    \item[] \dimeCheckBox{The tracking system updates automatically with new data, rather than requiring manual inputs.}
\end{itemize}



\section*{Survey Progress Checks}
\subsection*{Response Rates}

\begin{itemize}
    \item[] \dimeCheckBox{The total number of surveys attempted, completed, and declined is reported.}
    \item[] \dimeCheckBox{Response rates are disaggregated by day and enumerator.}
    \item[] \dimeCheckBox{Decline reasons are categorized and summarized.}
\end{itemize}

\subsection*{Interview Duration}

\begin{itemize}
    \item[] \dimeCheckBox{The average duration of interviews is reported.}
    \item[] \dimeCheckBox{Interview duration is broken down by module/section.}
    \item[] \dimeCheckBox{Unusually short or long interviews are flagged for review.}
\end{itemize}



\section*{Survey Quality Checks}
\subsection*{Duplicate \& Consistency Checks}

\begin{itemize}
    \item[] \dimeCheckBox{Duplicate entries in ID variables are identified and flagged.}
    \item[] \dimeCheckBox{If duplicates exist, non-ID variables are checked for differences.}
\end{itemize}

\subsection*{Missing Data \& Outlier Checks}

\begin{itemize}
    \item[] \dimeCheckBox{Missing values for key variables are reported.}
    \item[] \dimeCheckBox{Extended missing responses (\texttt{Do not know}, \texttt{Refuse to answer}) are summarized.}
    \item[] \dimeCheckBox{Continuous variables are checked for outliers and extreme values.}
\end{itemize}

\subsection*{Inconsistency Checks}

\begin{itemize}
    \item[] \dimeCheckBox{Inconsistent responses across related questions are flagged.}
    \item[] \dimeCheckBox{Inconsistencies over time (e.g., same respondent providing different answers in different waves) are reported.}
\end{itemize}

\subsection*{Survey Form Version Checks}

\begin{itemize}
    \item[] \dimeCheckBox{The number of responses is reported by form version and date.}
    \item[] \dimeCheckBox{Any discrepancies in form versions are flagged.}
\end{itemize}


\section*{Enumerator Performance Checks}
\subsection*{Response Rates by Enumerator}

\begin{itemize}
    \item[] \dimeCheckBox{The number/share of surveys attempted, completed, and declined is reported per enumerator.}
    \item[] \dimeCheckBox{The number of attempts per respondent is tracked per enumerator.}
\end{itemize}

\subsection*{Interview Duration by Enumerator}

\begin{itemize}
    \item[] \dimeCheckBox{The average interview duration is reported for each enumerator.}
    \item[] \dimeCheckBox{The average duration per module is reported for each enumerator.}
\end{itemize}

\subsection*{Enumerator-Specific Missing Data \& Outliers}

\begin{itemize}
    \item[] \dimeCheckBox{Overall missing rates are reported by enumerator and question type.}
    \item[] \dimeCheckBox{Average missing rates for key variables (which may lead to skipped sections) are reported per enumerator.}
    \item[] \dimeCheckBox{Outlier distributions are reported for each key variable per enumerator.}
\end{itemize}

\subsection*{Inconsistency \& Error Checks by Enumerator}

\begin{itemize}
    \item[] \dimeCheckBox{Total number of inconsistent responses is reported for each enumerator.}
    \item[] \dimeCheckBox{The average number of inconsistencies per day is tracked.}
    \item[] \dimeCheckBox{Total flagged errors per enumerator are summarized.}
    \item[] \dimeCheckBox{The average number of flagged errors per day is reported.}
\end{itemize}
